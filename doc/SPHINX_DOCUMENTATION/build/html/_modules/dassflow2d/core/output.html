[docs]    def get_res(self):
        """
        load out from result_xxx.file
        
        defines dictionary with callable items (x, y, bahty, h, zs, manningALPHA, u, v)
        	dict["x"].shape == (number of cell, number of writing time step)
        """
        def _my_grep(l, s):
            return [i for i in l if re.search(s, i)]
            
        

        def _source_w_gnuplot(pathfile):
            res = pd.read_table(pathfile, skiprows=2, header = None, sep ='\s+')
            res = res.rename({0:'id_pixel', 1: 'x', 2: 'y', 3: 'bathy', 4: 'h', 5: 'zs', 6: 'manningALPHA', 7: 'u', 8: 'v'}, axis='columns')
            res = res.drop(['id_pixel'], axis=1)
            for colname in list(res.columns):
                try:
                    res[colname] =   res[colname].astype(float)
                except ValueError as e:
                    #print("error",e,"on variable",colname)
                    #print("real value is",res[colname])
                    #print( "result is set to zero")
                    res[colname] = 0
            return(res)
            
        # list all files in res_dir
        onlyfiles = [f for f in os.listdir(self.res_dir) if os.path.isfile(os.path.join(self.res_dir, f))]


        # get .dat files
        
        # source all files and sort them by ascending time
        all_files = _my_grep(onlyfiles, ".dat")
        all_times = np.empty(shape = len(all_files))

        for id_file in range(len(all_files)):
            tmp = all_files[id_file].split("_")[1]
            tmp = re.sub(".dat", "", tmp)
            if tmp == "initial":
                tmp = 0
            if tmp == "final":
                tmp = 9999999999999999
            all_times[id_file] = tmp

        id_reorder = np.argsort(all_times)
        all_times = all_times[id_reorder]
        all_files = [all_files[i] for i in id_reorder]
        
        
        all_res = dict()

        for i in range(len(all_files)):
            all_res[all_times[i]] = _source_w_gnuplot(self.res_dir + all_files[i])
            
            
        return(all_res, all_times)



[docs]    def save_res(self, hdf5_path):
        """
        save output from result_xxx.file t ohdf5 file
        
        **parameters**
        
        hdf5_path : path to hdf5 file to save
        """
        f = h5py.File(hdf5_path, "a")
        
        
        nb_timestep =  len(self.all_res)
        nb_cell = len(self.all_res[0])

        h = np.empty(shape = (nb_cell, nb_timestep) )
        u = np.empty(shape = (nb_cell, nb_timestep) )
        v = np.empty(shape = (nb_cell, nb_timestep) )
        bathy = np.empty(shape = (nb_cell, nb_timestep) )
        manning= np.empty(shape = (nb_cell, nb_timestep) )

        for id_time in range(len(self.all_res)):
            my_res = self.all_res[ self.all_times[id_time] ]
            h[ :, id_time] = my_res["h"].to_numpy()
            u[ :, id_time] = my_res["u"].to_numpy()
            v[ :, id_time] = my_res["v"].to_numpy()
            bathy[ :, id_time] = my_res["bathy"].to_numpy()
            manning[ :, id_time] = my_res["manningALPHA"].to_numpy()


        output = f.create_group("output")
        spatial_var = output.create_group("spatial_var")

        h_new = spatial_var.create_dataset("h", (nb_cell,nb_timestep), 'f')
        spatial_var["h"][:,:]= h

        u_new = spatial_var.create_dataset("u", (nb_cell,nb_timestep), 'f')
        spatial_var["u"][:,:]= u

        v_new = spatial_var.create_dataset("v", (nb_cell,nb_timestep), 'f')
        spatial_var["v"][:,:] = v

        bathy_new = spatial_var.create_dataset("bathy", (nb_cell,nb_timestep), 'f')
        spatial_var["bathy"][:,:] = bathy

        manning_alpha_new = spatial_var.create_dataset("manning_alpha", (nb_cell,nb_timestep), 'f')
        spatial_var["manning_alpha"][:,:] = manning


        time_new = output.create_dataset("time",  (nb_timestep), 'f')
        output["time"][:] = self.all_times


        f.attrs['package version'] = version('dassflow2d')
        f.close()
